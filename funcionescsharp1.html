Funciones simuladas estilo C# con múltiples entradas

1. Concatenar nombres (string nombre1, string nombre2)

return nombre1 + " " + nombre2;

Resultado: "Juan Pérez"

2. Obtener iniciales (string nombreCompleto)

string[] partes = nombreCompleto.Split(" ");
return partes[0][0] + "." + partes[1][0] + ".";

Resultado: "J.P."

3. Validar contraseña (string input)

if (input.Length < 8 || !input.Contains("1")) return "Contraseña inválida";
return "Contraseña válida";

Error: Contraseña inválida

4. Repetir palabra (string palabra, int veces)

string resultado = "";
for (int i = 0; i < veces; i++) resultado += palabra + " ";
return resultado;

Resultado: "Hola Hola Hola "

5. Comprobar subcadena (string texto, string buscar)

if (!texto.Contains(buscar)) return "No encontrado";
return "Subcadena presente";

Resultado: "Subcadena presente"

6. Invertir cadena (string texto)

char[] arr = texto.ToCharArray();
Array.Reverse(arr);
return new string(arr);

Resultado: "odnum"

7. Sumar números en texto (string valores)

string[] nums = valores.Split(',');
int suma = 0;
foreach (var n in nums) suma += int.Parse(n);
return suma;

Resultado: 45

8. Longitud total de palabras (string[] palabras)

int total = 0;
foreach (string p in palabras) total += p.Length;
return total;

Resultado: 17

9. Evaluar frase vacía (string texto)

if (string.IsNullOrWhiteSpace(texto)) return "Texto vacío o en blanco";
return "Texto válido";

Error: Texto vacío o en blanco

10. Contar apariciones (string texto, char caracter)

int contador = 0;
foreach (char c in texto)
{ if (c == caracter) contador++; }
return contador;

Resultado: 3
