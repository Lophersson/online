1. Concatenar elementos de un array separándolos por coma

string ConcatenarArray(string[] arr) { string resultado = ""; for(int i = 0; i < arr.Length; i++) { resultado += arr[i]; if(i < arr.Length - 1) resultado += ", "; } return resultado; }

2. Contar palabras que contienen una subcadena

int ContarPalabrasConSubcadena(string texto, string sub) { string[] palabras = texto.Split(' '); int count = 0; foreach(var p in palabras) { if(p.Contains(sub)) count++; } return count; }
EntradaSalida
"hola mundo hola", "lo"2
"programar en C# es divertido", "r"2
"abracadabra", "abra"1
"uno dos tres cuatro", "cinco"0

6. Invertir palabra usando bucle

string Invertir(string texto) { string resultado = ""; for(int i = texto.Length - 1; i >= 0; i--) { resultado += texto[i]; } return resultado; }
EntradaSalida

"12345""54321"
"a""a"

8. Obtener número de repeticiones consecutivas de un carácter

int RepeticionesConsecutivas(string texto, char c) { int maxCount = 0, count = 0; foreach(char ch in texto) { if(ch == c) { count++; if(count > maxCount) maxCount = count; } else { count = 0; } } return maxCount; }
EntradaSalida
"aaabbbaa", 'a'3
"bbbccc", 'b'3
"abcde", 'f'0
"", 'a'0
